6   import static java.util.Arrays.asList;

8   /**
9    * Simple class with a main method to call from unit tests
10   */
11  public class Slf4jMain
12  {
13      /**
14       * Print on stdout the logger class (slf4j binding) and then
15       * logs at info level args with Hello as suffix.
16       * 
17       * @param args the arguments
18       */
19      public static void main( String... args )
20      {
21          final Logger logger = LoggerFactory.getLogger( Slf4jMain.class );
22          System.out.println( logger.getClass().getName() );
23          logger.info( "hello" + asList( args ) );
24      }
25  }
